" date: 2025-09-26 14:51:30

" source ~/.vimrc
" ==============================================================================
" 模式/映射
" map 映射；nore 不递归；un 取消；clear 清除
" i 插入；n 普通；v 可视
" ==============================================================================

" ==============================================================================
" # vim start
" leader键
let mapleader=' '
" 设置行号
set number
" 设置相对行号
set relativenumber
" 上下预留5行代码
set so=5
" 使用系统剪切板
set clipboard=unnamed
" 默认按下Esc后,需要等待1秒才生效,设置Esc超时时间为100ms,尽快生效
set ttimeoutlen=100
" 提示静音
set vb t_vb=
" 高亮显示所有搜索到的内容
set hlsearch
" 光标立刻跳转到搜索到内容
set incsearch
" 搜索到最后匹配的位置后再次搜索不回到第一个匹配处
set nowrapscan
" 搜时忽略大小写，但在有一个或以上大写字母时仍保持对大小写敏感
set ignorecase smartcase
" 自动缩进
set autoindent
" 使用 << and >> 缩进长度为2
set shiftwidth=2
" tab 的长度为2
set tabstop=2
" 退格键一次可以删除2个空格
set softtabstop=2
" tab替换为空格
set expandtab
" # vim end
" ==============================================================================

" ==============================================================================
