## 二分查找

对已排序的整数数据进行查找。

```typescript
function binSearch(arr: number[], searchValue: number) {
  let leftPoint = 0
  let rightPoint = arr.length - 1
  while (leftPoint <= rightPoint) {
    let midPoint = ~~((leftPoint + rightPoint) / 2)
    if (arr[midPoint] < searchValue) {
      leftPoint = midPoint + 1
    } else if (arr[midPoint] > searchValue) {
      rightPoint = midPoint - 1
    } else {
      return midPoint
    }
  }
  return -1
}
```
